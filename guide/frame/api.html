# 与服务端交互

此框架与服务端交互是通过自定义的http.js文件。

http.js是新增的文件，用于页面的所有请求。下面主要介绍一下该文件中的post方法，postReport方法，checkStatus方法，checkCode方法，之后将举一个具体的调用示例。

  • 目录介绍：
├── utils                       
│   ├── http.js                        # 请求文件
  • post方法：
    post方法用于页面的绝大部分请求，下面来详细介绍下该方法：
import axios from 'axios'
import store from '@/store/index'
post({
  url,
  data
}) {
  const sessionid = store.state.user.UserUserId   // 存在store里面的userId
  const sessionstr = store.state.user.UserSessionstr // 存在store里面的sessionStr
  const Authorization = sessionid + ',' + sessionstr + ',' + url
  if (sessionid != null && sessionid !== '') { // 判断userId和sessionStr是否为空
    // userId和sessionStr不为空，在请求头中写入userId和sessionStr
    data.userStr = sessionid
    data.sessionStr = sessionstr
    // 将axios return 出去
    return axios({
      method: 'post', // 请求方式
      baseURL: window.g.APP_URL_PATH, // 全局路径
      url: url, // 具体接口路径
      data: data, // 请求参数
      dataType: 'json', // 预期服务器返回的数据类型
      headers: {
        Authorization, // Authorization 服务端要求，包含userId、sessionStr和具体接口路径
        'Session-Str': sessionstr, // sessionStr
        'User-Id': sessionid, // userId
      },
      timeout: 120000 // 超时时间
    })
      .then(response => {
        return checkStatus(response) // 检查服务端返回的状态码
      })
      .then(res => {
        return checkCode(res) // 检查服务端返回的code值
      })
  } else {
    // userId和sessionStr为空
    return axios({
      method: 'post',
      baseURL: window.g.APP_URL_PATH,
      url: url,
      data: data,
      dataType: 'json',
      timeout: 120000
    })
      .then(response => {
        return checkStatus(response)
      })
      .then(res => {
        return checkCode(res)
      })
  }
}

方法介绍

  • 该方法的请求方式method为post。
  • 该方法接收两个参数url和data，url是请求的路径，data是请求的参数。
  • 传入这两个参数后，从store里面查找用户的userId和sessionStr，如果存在则在请求头中加入userId和sessionStr，以便后台验证session。
  • postReport方法：

该方法用于导出文件，与post方法大致相同，唯一不同的是参数中新增了一个返回值类型responseType: 'blob'，后台返回的数据会被强制转为blob类型。

import axios from 'axios'
import store from '@/store/index'
postReport({
  url,
  data
}) {
  const sessionid = store.state.user.UserUserId
  const sessionstr = store.state.user.UserSessionstr
  const Authorization = sessionid + ',' + sessionstr + ',' + url
  if (sessionid != null && sessionid !== '') {
    data.userStr = sessionid
    data.sessionStr = sessionstr
    return axios({
      responseType: 'blob', // 返回值类型：后台返回的数据会被强制转为blob类型
      method: 'post',
      baseURL: window.g.APP_URL_PATH,
      url: url,
      data: data,
      dataType: 'json',
      headers: {
        Authorization,
        'Session-Str': sessionstr,
        'User-Id': sessionid
      },
      timeout: 120000
    })
      .then(response => {
        return checkStatus(response)
      })
      .then(res => {
        return checkCode(res)
      })
  } else {
    return axios({
      method: 'post',
      baseURL: window.g.APP_URL_PATH,
      url: url,
      data: data,
      dataType: 'json',
      timeout: 120000
    })
      .then(response => {
        return checkStatus(response)
      })
      .then(res => {
        return checkCode(res)
      })
  }
}
  • checkStatus方法：

checkStatus方法主要用于检查服务器端返回的状态码，在客户端发送post请求获得服务端响应时，先去检查服务端响应的状态码。

function checkStatus(response) {
  if (response === undefined) {
    // 异常
  } else {
    if (
      response &&
      (response.status === 200 ||
        response.status === 304 ||
        response.status === 400 ||
        response.status === 500 ||
        response.status === 504 ||
        response.status === 404)
    ) {
      return response
      // 如果不需要除了data之外的数据，可以直接 return response.data
    }
    // 异常状态下，把错误信息返回去
    return {
      status: -404,
      data: {
        code: -44444,
        message: '异常'
      }
    }
  }
}

方法介绍

  • 如果返回的response为undefined则不做任何处理。
  • 如果返回了response，且response的值在200,304,400,500,504,404中。则将response返回。
  • 如果返回的状态码不在以上值中，则返回状态-404，code值-44444。
  • checkCode方法：

该方法为了检验后端返回的code值，在异常状态下抛出报错信息，并且退回到登录页面，多为会话失效时用到。

// 防抖使用的变量
import Vue from 'vue'
let checkCodeTimer = true
function checkCode(res) {
  if (res.data) {
    // 服务端返回的异常信息，一般用于会话失效
    if (res.data.code === -44444 || res.data.code === -44445 || res.data.status === 500) {
      // 防抖，三秒内里面的内容多次请求也只会执行一次
      if (checkCodeTimer) {
        checkCodeTimer = false
        Vue.prototype.$message({
          type: 'warning',
          message: res.data.message,
          duration: 2500
        })
        setTimeout(() => {
          checkCodeTimer = true
          // 强制退出登录清空数据
          store.dispatch('LogOut').then(() => {
            // 跳转到登录页
            global.router.push('/login')
          })
        }, 3000)
      }
    } else {
      return res
    }
  } else {
    return res
  }
}

方法介绍

  • 如果服务端返回的code值为-44444或-44445，或者返回的status是500时，判定为异常状态。给客户端抛出异常，并请求退出登录接口清空userId和sessionStr后返回到登录页面。
  • 这里有一个小点要注意下，用户进入一个页面会同时请求多个接口，当用户的登录状态为异常状态时，每一个接口都会抛出异常信息，所以需要使用checkCodeTimer防止页面同时弹出多个报错信息。原理是当返回第一个报错信息时，将checkCodeTimer状态值改变，因此其余的报错信息就不会在页面上弹出。此方法目的主要是优化用户体验。
  • 发送请求：
├── src                       
│   ├── api                        # api文件夹
│   │   ├── protocol.js            # 所有请求

所有的与服务端联调的接口都写在了protocol.js文件中。
下面来举一个发送请求的示例：

// protocol.js文件
// 会员管理 -> 会员列表查询
param_findUserList: {
  url: '/console/member/queryMemberInfo',
  data: {
    userNo: null, // 会员id
    mobile: null, // 手机号
    siginNickname: null, // 登录账号
    email: null, // 邮箱
    unlimited: null, // 不限
    page: 1, // 第几页
    size: 15, // 每页搜索结果个数
    sort: [{
      direction: 'DESC',
      property: 'createTime'
    }]
  }
}

import http from './utils/http'
// 具体请求的文件
http.post(this.protocol.param_findUserList).then(res => {
  const { code, message, value } = res.data
})

方法介绍

  • 将http.js文件import进来（注意：实际项目中http.js文件会在整体中的main.js文件中引入，无需单独引入，直接使用this.http即可。）
  • http.js文件中导出的方法包含post，所以可以用http.post直接调用该方法。
  • 将参数传入post方法中，参数中包含url和data。
  • 使用.then()，post会在执行完内部的checkStatus()和checkCode()方法后，执行then方法后面的内容。
